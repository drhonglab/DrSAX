var DSX = new DSX;

var osc = new DSX.Osc("sawtooth", 700);
var gain = new DSX.Amp({gain: 0.5});

osc.connect(gain);
gain.connect(DAC);


var oscfrequency = new DSX.valueChange("knob1", osc.frequency);
var ampgain = new DSX.valueChange("knob2", gain.gain);
var sound_on_off = new DSX.valueChange("switch1");

sound_on_off.setRange(switch1);

function switch1(e) {
    if (e.target.value == "1") {

        osc.start();
    }
    if (e.target.value == "0") {

        osc.stop();
    }
}
